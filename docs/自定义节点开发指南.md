## 基本概念

### LogicFlow 节点架构

LogicFlow 采用 **MVVM** 架构设计：

- **Model（模型）**: 管理节点的数据和状态
- **View（视图）**: 负责节点的渲染和交互

### 内置节点类型

| 节点类型 | Model | View | 说明 |
|---------|-------|------|------|
| 矩形节点 | `RectNodeModel` | `RectNode` | 矩形/圆角矩形 |
| 圆形节点 | `CircleNodeModel` | `CircleNode` | 圆形 |
| 菱形节点 | `DiamondNodeModel` | `DiamondNode` | 菱形 |
| 椭圆节点 | `EllipseNodeModel` | `EllipseNode` | 椭圆 |
| 多边形节点 | `PolygonNodeModel` | `PolygonNode` | 多边形 |
| HTML节点 | `HtmlNodeModel` | `HtmlNode` | 自定义HTML |

## 自定义节点流程

### 步骤1: 创建节点模型

```typescript
// MyNodeModel.ts
import { RectNodeModel } from '@logicflow/core';
import type { GraphModel } from '@logicflow/core';
import type LogicFlow from '@logicflow/core';

// 定义节点属性类型
export interface IMyNodeProperties {
  width?: number;
  height?: number;
  color?: string;
  [key: string]: unknown;
}

export class MyNodeModel extends RectNodeModel<IMyNodeProperties> {
  // 节点类型标识
  static nodeTypeName = 'myNode';

  constructor(data: LogicFlow.NodeConfig<IMyNodeProperties>, graphModel: GraphModel) {
    super(data, graphModel);
    this.initNodeData(data);
    this.setAttributes();
  }

  setAttributes() {
    super.setAttributes();
    // 从 properties 读取自定义属性
    const { width, height, color } = this.properties;
    if (width) this.width = width;
    if (height) this.height = height;
  }

  getNodeStyle() {
    const style = super.getNodeStyle();
    return {
      ...style,
      fill: 'transparent',
      stroke: this.properties.color || '#333',
      strokeWidth: 2,
    };
  }

  getDefaultAnchor() {
    const { x, y, width, height } = this;
    return [
      { x, y: y - height / 2, id: `${this.id}_0` },
      { x: x + width / 2, y, id: `${this.id}_1` },
      { x, y: y + height / 2, id: `${this.id}_2` },
      { x: x - width / 2, y, id: `${this.id}_3` },
    ];
  }
}
```

### 步骤2: 创建节点视图

```typescript
// MyNode.tsx
import { h, RectNode } from '@logicflow/core';
import type { GraphModel } from '@logicflow/core';
import type MyNodeModel from './MyNodeModel';

export type IMyNodeProps = {
  model: MyNodeModel;
  graphModel: GraphModel;
};

export class MyNode extends RectNode<IMyNodeProps> {
  getShape() {
    const { model } = this.props;
    const { x, y, width, height } = model;
    const style = model.getNodeStyle();

    return h(
      'g',
      {
        transform: `translate(${x - width / 2}, ${y - height / 2})`,
      },
      // 绘制矩形
      h('rect', {
        x: 0,
        y: 0,
        width,
        height,
        fill: style.fill,
        stroke: style.stroke,
        strokeWidth: style.strokeWidth,
        rx: 4,
        ry: 4,
      }),
      // 绘制图标或其他元素
      h('text', {
        x: width / 2,
        y: height / 2,
        textAnchor: 'middle',
        dominantBaseline: 'middle',
        fill: style.stroke,
        fontSize: 14,
      }, 'MyNode'),
    );
  }
}
```

### 步骤3: 注册节点

```typescript
// index.ts
import type LogicFlow from '@logicflow/core';
import { MyNode } from './MyNode';
import { MyNodeModel } from './MyNodeModel';

export const MY_NODE_TYPE = 'myNode';

export function registerMyNode(lf: LogicFlow, config?: { defaultWidth?: number; defaultHeight?: number }) {
  class CustomMyNodeModel extends MyNodeModel {
    constructor(data: any, graphModel: any) {
      data.properties = {
        width: config?.defaultWidth || 100,
        height: config?.defaultHeight || 60,
        ...data.properties,
      };
      super(data, graphModel);
    }
  }

  lf.register({
    type: MY_NODE_TYPE,
    view: MyNode,
    model: CustomMyNodeModel,
  });
}

export function createMyNodeConfig(x: number, y: number, properties?: Record<string, unknown>) {
  return {
    type: MY_NODE_TYPE,
    x,
    y,
    properties,
  };
}
```

## 基于 RectNode 扩展节点

### 示例：排他网关节点

排他网关是 BPMN 规范中的网关节点，渲染为菱形并带有 X 图标。

#### 模型实现

```typescript
// ExclusiveGatewayModel.ts
import { cloneDeep, isNil } from 'lodash-es';
import { RectNodeModel } from '@logicflow/core';
import type { GraphModel } from '@logicflow/core';
import type LogicFlow from '@logicflow/core';

const DEFAULT_CONFIG = {
  defaultWidth: 50,
  defaultHeight: 50,
  defaultColor: '#E6A23C',
};

export class ExclusiveGatewayModel extends RectNodeModel {
  static nodeTypeName = 'exclusiveGateway';

  constructor(data: LogicFlow.NodeConfig, graphModel: GraphModel) {
    super(data, graphModel);
    this.initNodeData(data);
    this.setAttributes();
  }

  initNodeData(data: LogicFlow.NodeConfig) {
    super.initNodeData(data);
    if (isNil(this.width)) this.width = DEFAULT_CONFIG.defaultWidth;
    if (isNil(this.height)) this.height = DEFAULT_CONFIG.defaultHeight;
  }

  setAttributes() {
    super.setAttributes();
    const { width, height } = this.properties;
    if (!isNil(width)) this.width = width;
    if (!isNil(height)) this.height = height;
  }

  getNodeStyle() {
    const style = super.getNodeStyle();
    const color = this.properties.color || DEFAULT_CONFIG.defaultColor;
    return {
      ...style,
      fill: 'transparent',
      stroke: color,
      strokeWidth: 2,
    };
  }

  getDefaultAnchor() {
    const { x, y, width, height } = this;
    return [
      { x, y: y - height / 2, id: `${this.id}_0` },
      { x: x + width / 2, y, id: `${this.id}_1` },
      { x, y: y + height / 2, id: `${this.id}_2` },
      { x: x - width / 2, y, id: `${this.id}_3` },
    ];
  }

  getIconColor() {
    return this.properties.color || DEFAULT_CONFIG.defaultColor;
  }
}
```

#### 视图实现

```typescript
// ExclusiveGatewayNode.tsx
import { h, RectNode } from '@logicflow/core';
import type { GraphModel } from '@logicflow/core';
import type ExclusiveGatewayModel from './ExclusiveGatewayModel';

const EXCLUSIVE_ICON = {
  diamondPath: 'M58.585789,28.585789L31.414215,1.4142134L30.000002,0L28.585789,1.4142133...',
  innerBorderPath: 'M57.171577,30.000002L30.000002,2.8284268L2.8284268,30.000002...',
  xIconVertical: 'M29.170242,19.684513L30.829813,19.684513L30.829813,40.226351...',
  xIconHorizontal: 'M19.876919,29.138229L40.123088,29.138229L40.123088,30.762846...',
};

export class ExclusiveGatewayNode extends RectNode {
  getShape() {
    const { model } = this.props;
    const { x, y, width, height } = model;
    const style = model.getNodeStyle();
    const iconColor = model.getIconColor();
    const scaleX = width / 60;
    const scaleY = height / 60;

    return h('g', { transform: `translate(${x - width / 2}, ${y - height / 2})` },
      h('path', {
        d: EXCLUSIVE_ICON.diamondPath,
        fill: style.fill || 'transparent',
        stroke: style.stroke || iconColor,
        strokeWidth: style.strokeWidth || 2,
        transform: `scale(${scaleX}, ${scaleY})`,
        transformOrigin: '30px 30px',
      }),
      h('path', {
        d: EXCLUSIVE_ICON.xIconVertical,
        fill: iconColor,
        transform: `scale(${scaleX}, ${scaleY})`,
        transformOrigin: '30px 30px',
      }),
      h('path', {
        d: EXCLUSIVE_ICON.xIconHorizontal,
        fill: iconColor,
        transform: `scale(${scaleX}, ${scaleY})`,
        transformOrigin: '30px 30px',
      }),
    );
  }
}
```

## 最佳实践

### 1. 避免使用 mobx 装饰器

```typescript
// ❌ 不推荐
import { observable } from 'mobx';

class MyModel extends RectNodeModel {
  @observable customProp = 'value';
}

// ✅ 推荐
class MyModel extends RectNodeModel {
  getCustomProp() {
    return this.properties.customProp || 'default';
  }
}
```

### 2. 使用 @logicflow/core 的 h 函数

```typescript
// ❌ 不推荐
import { createElement as h } from 'preact/compat';

// ✅ 推荐
import { h, RectNode } from '@logicflow/core';
```

### 3. 属性通过 properties 传递

```typescript
// 添加节点时
lf.addNode({
  type: 'myNode',
  x: 100,
  y: 100,
  properties: {
    color: '#52c41a',
    width: 80,
    height: 80,
  },
});

// 模型中读取
getNodeStyle() {
  const color = this.properties.color || '#333';
  return { stroke: color };
}
```

### 4. 提供工厂函数

```typescript
// 便于创建节点配置
export function createMyNode(x: number, y: number, properties?: Partial<IMyNodeProperties>) {
  return {
    type: MY_NODE_TYPE,
    x,
    y,
    properties: {
      width: 100,
      height: 60,
      ...properties,
    },
  };
}

// 使用
lf.addNode(createMyNode(100, 100, { color: '#1890ff' }));
```

### 5. 动态修改属性

```typescript
// 获取节点模型
const node = lf.getNodeModelById('node_id');

// 修改属性
node.setProperties({
  ...node.properties,
  color: '#ff4d4f',
});
```
